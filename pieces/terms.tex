\newglossaryentry{HER2- }{
  name={HER2- },
  sort={HER2- },
  description={ refers to a classification of cancer cells that do not overexpress the HER2 protein, a receptor involved in cell growth and differentiation. In HER2-negative cancers, particularly in breast cancer, HER2-targeted therapies are usually ineffective, so treatment options focus on other pathways. HER2 status is a crucial factor in determining treatment strategies and prognosis.}
}
\newglossaryentry{Overall Survival }{
  name={Overall Survival },
  sort={Overall Survival },
  description={ is the duration from the start of treatment or diagnosis until death from any cause. It is a key endpoint in clinical trials, reflecting the effectiveness of a treatment in extending the life of patients. OS is considered a definitive measure of clinical benefit, providing a comprehensive view of a treatment's impact on patient survival.}
}

\newglossaryentry{Progression-Free Survival }{
  name={Progression-Free Survival},
  sort={Progression-Free Survival},
  description={is the length of time during and after treatment in which a patient's disease does not worsen or progress. It is an important endpoint in clinical trials, particularly for cancer therapies, as it indicates how well a treatment can delay disease advancement. PFS is used to assess the efficacy of new treatments in controlling the disease.}
}
\newglossaryentry{Eastern Cooperative Oncology Group (ECOG) Scale}{
  name={Eastern Cooperative Oncology Group},
  sort={Eastern Cooperative Oncology Group},
  description={also known as the ECOG Performance Status, is a standardized measure used to assess a cancer patient's level of functioning and ability to perform daily activities. The scale ranges from 0 (fully active, no restrictions) to 5 (dead). It helps guide treatment decisions and assess the patient's ability to tolerate specific treatments.}
}

\newglossaryentry{CDK4/6i (Cyclin-Dependent Kinases 4 and 6 Inhibitors)}{
  name={CDK4/6i (Cyclin-Dependent Kinases 4 and 6 Inhibitors)},
  sort={CDK4/6i (Cyclin-Dependent Kinases 4 and 6 Inhibitors)},
  description={are a class of targeted cancer therapies that inhibit the activity of cyclin-dependent kinases 4 and 6. These kinases play a crucial role in cell cycle progression, and their inhibition can prevent cancer cells from proliferating. CDK4/6 inhibitors are commonly used in the treatment of hormone receptor-positive, HER2-negative advanced breast cancer to slow disease progression and improve outcomes.}
}
